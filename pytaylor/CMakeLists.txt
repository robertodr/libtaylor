# define minimum cmake version
cmake_minimum_required(VERSION 3.5 FATAL_ERROR)

# project name and supported language
project(pytaylor LANGUAGES CXX)

# require C++11
set(CMAKE_CXX_STANDARD 11)
set(CMAKE_CXX_EXTENSIONS OFF)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

# the pybind11 package will provide the pybind11::module imported target
find_package(pybind11 CONFIG REQUIRED)

# create python module
add_library(pytaylor
  MODULE
    pytaylor.cpp
  )

if(WIN32 AND (NOT MSVC))
  target_compile_definitions(pytaylor
    PUBLIC
      "_hypot=hypot"
    )
endif()

target_include_directories(pytaylor
  PUBLIC
    ${LIBTAYLOR_INCLUDE_DIR}
  )

target_link_libraries(pytaylor
  PUBLIC
    pybind11::module
  )

set_target_properties(pytaylor
  PROPERTIES
    PREFIX "${PYTHON_MODULE_PREFIX}"
    SUFFIX "${PYTHON_MODULE_EXTENSION}"
  )

file(
  COPY
    __init__.py
    version.py
  DESTINATION
    ${CMAKE_CURRENT_BINARY_DIR}
  )

install(
  TARGETS
    pytaylor
  LIBRARY DESTINATION .
  )
